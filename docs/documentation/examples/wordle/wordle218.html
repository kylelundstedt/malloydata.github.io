WITH __stage0 AS (
  SELECT 
     upper(words.word) as word
  FROM `malloy-data.malloytest.words_bigger` as words
  WHERE ((length(words.word))=5)and REGEXP_CONTAINS(words.word, r'^[a-z]{5}$')
)
, __stage1 AS (
  SELECT
    group_set,
    CASE WHEN group_set IN (0,1) THEN
      base.word
      END as word__0,
    CASE WHEN group_set=1 THEN
      substr(base.word,numbers_0.num,1)
      END as letter__1,
    CASE WHEN group_set=1 THEN
      numbers_0.num
      END as position__1
  FROM __stage0 as base
  LEFT JOIN `malloy-data.malloytest.numbers` AS numbers_0
    ON 1=1 AND (numbers_0.num<=5)
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2,3,4
)
, __stage2 AS (
  SELECT
    word__0 as word,
    ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(
      letter__1 as letter, 
      position__1 as position
      ) END IGNORE NULLS  ORDER BY  position__1 ASC ) as letters
  FROM __stage1
  GROUP BY 1
  ORDER BY 1 asc
)
, __stage3 AS (
  SELECT
    group_set,
    CASE WHEN group_set IN (0,1) THEN
      letters_0.letter
      END as letter__0,
    CASE WHEN group_set IN (0,1) THEN
      letters_0.position
      END as position__0,
    CASE WHEN group_set=0 THEN
      COUNT(DISTINCT wordle.word)
      END as word_count__0,
    CASE WHEN group_set=1 THEN
      wordle.word
      END as word__1
  FROM __stage2 as wordle
  LEFT JOIN UNNEST(wordle.letters) as letters_0
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  WHERE (COALESCE(NOT(REGEXP_CONTAINS(wordle.word, r'(S|ED)$')),FALSE))
  AND (COALESCE(NOT(REGEXP_CONTAINS(wordle.word, r'(S|ED)$')),FALSE))
  GROUP BY 1,2,3,5
)
, __stage4 AS (
  SELECT
    letter__0 as letter,
    position__0 as position,
    ANY_VALUE(CASE WHEN group_set=0 THEN word_count__0 END) as word_count,
    ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(
      word__1 as word
      ) END IGNORE NULLS  ORDER BY  word__1 asc ) as words_list
  FROM __stage3
  GROUP BY 1,2
  ORDER BY 3 desc
)
SELECT 
   words_list_0.word as word,
   COALESCE(CAST((
    (
      SUM(DISTINCT
        (CAST(ROUND(COALESCE(base.word_count,0)*(1*1.0), 9) AS NUMERIC) +
        (cast(cast(concat('0x', substr(to_hex(md5(CAST(base.__distinct_key AS STRING))), 1, 15)) as int64) as numeric) * 4294967296 + cast(cast(concat('0x', substr(to_hex(md5(CAST(base.__distinct_key AS STRING))), 16, 8)) as int64) as numeric)) * 0.000000001
      ))
      -
       SUM(DISTINCT (cast(cast(concat('0x', substr(to_hex(md5(CAST(base.__distinct_key AS STRING))), 1, 15)) as int64) as numeric) * 4294967296 + cast(cast(concat('0x', substr(to_hex(md5(CAST(base.__distinct_key AS STRING))), 16, 8)) as int64) as numeric)) * 0.000000001)
    )/(1*1.0)) as FLOAT64),0) as score
FROM (SELECT GENERATE_UUID() as __distinct_key, * FROM __stage4 as x) as base
LEFT JOIN UNNEST(base.words_list) as words_list_0
GROUP BY 1
ORDER BY 2 desc

      wordle.word

      word__1 as word

FROM (SELECT GENERATE_UUID() as __distinct_key, * FROM __stage4 as x) as base
