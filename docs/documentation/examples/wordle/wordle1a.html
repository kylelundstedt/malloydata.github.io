Letters and Positioning.

The query below produces a table with the numbers 1 to 5

source: numbers is table('malloy-data.malloytest.numbers') {
  where: num <= 5
}
query: numbers -> { project: num }

SELECT 
   numbers.num as num
FROM `malloy-data.malloytest.numbers` as numbers
WHERE numbers.num<=5

Cross join these two tables to produce letter positioning.

The result is a table with nested data. Each word contains a sub-table with a letter in each position.

// define the query
query: words_and_position is from(words -> five_letter_words) {
  // cross join is missing at the moment
  join_cross: numbers
} -> {
  group_by: word
  nest: letters is {
    order_by: 2
    group_by:
      letter is substr(word, numbers.num, 1)
      position is numbers.num
  }
}

WITH __stage0 AS (
  SELECT 
     upper(words.word) as word
  FROM `malloy-data.malloytest.words_bigger` as words
  WHERE ((length(words.word))=5)and REGEXP_CONTAINS(words.word, r'^[a-z]{5}$')
)
, __stage1 AS (
  SELECT
    group_set,
    CASE WHEN group_set IN (0,1) THEN
      base.word
      END as word__0,
    CASE WHEN group_set=1 THEN
      substr(base.word,numbers_0.num,1)
      END as letter__1,
    CASE WHEN group_set=1 THEN
      numbers_0.num
      END as position__1
  FROM __stage0 as base
  LEFT JOIN `malloy-data.malloytest.numbers` AS numbers_0
    ON 1=1 AND (numbers_0.num<=5)
  CROSS JOIN (SELECT row_number() OVER() -1  group_set FROM UNNEST(GENERATE_ARRAY(0,1,1)))
  GROUP BY 1,2,3,4
)
SELECT
  word__0 as word,
  ARRAY_AGG(CASE WHEN group_set=1 THEN STRUCT(
    letter__1 as letter, 
    position__1 as position
    ) END IGNORE NULLS  ORDER BY  position__1 ASC ) as letters
FROM __stage1
GROUP BY 1
ORDER BY 1 asc
